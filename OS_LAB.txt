1. Write a C program to create a child process. Display different messages in parent process and child process. Display PID and PPID of both parent and child process. Block parent process until child completes using wait system call.


#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>

int main() {
    pid_t pid;

    pid = fork();

    if (pid < 0) {
        printf("Fork failed\n");
        return 1;
    }
    else if (pid == 0) {
        // Child process
        printf("This is child process\n");
        printf("Child PID  = %d\n", getpid());
        printf("Child PPID = %d\n", getppid());
        printf("Child process completed\n");
        exit(0);
    }
    else {
        // Parent process
        wait(NULL);   // parent waits until child finishes
        printf("This is parent process\n");
        printf("Parent PID  = %d\n", getpid());
        printf("Parent PPID = %d\n", getppid());
        printf("Child with PID %d has finished\n", pid);
    }

    return 0;
}

3. Create a zombie (defunct) child process (a child with exit() call, but no corresponding wait() in the sleeping parent) and allow the init process to adopt it (after parent terminates). Run the process as background process and run ps command.

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

int main() {
    pid_t pid = fork();

    if (pid < 0) {
        printf("Fork failed\n");
        return 1;
    }
    else if (pid == 0) {
        printf("Child process\n");
        printf("Child PID = %d\n", getpid());
        printf("Child exiting...\n");
        exit(0);   // child exits immediately
    }
    else {
        printf("Parent process\n");
        printf("Parent PID = %d\n", getpid());
        printf("Sleeping for 20 seconds without wait()...\n");
        sleep(20);   // child becomes zombie during this time
        printf("Parent exiting now\n");
    }

    return 0;
}

4. Write a multithreaded program that performs different sorting algorithms. The program should work as follows: the user enters on the command line the number of elements to sort and the elements themselves. The program then creates separate threads, each using a different sorting algorithm. Each thread sorts the array using its corresponding algorithm and displays the time taken to produce the result. The main thread waits for all threads to finish and then displays the final sorted array.


#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <string.h>
#include <time.h>

typedef struct {
    int *arr;
    int n;
    char name[20];
} SortData;

void printArray(int arr[], int n) {
    for (int i = 0; i < n; i++)
        printf("%d ", arr[i]);
    printf("\n");
}

void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

void selectionSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int min = i;
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[min])
                min = j;
        }
        int temp = arr[i];
        arr[i] = arr[min];
        arr[min] = temp;
    }
}

void insertionSort(int arr[], int n) {
    for (int i = 1; i < n; i++) {
        int key = arr[i];
        int j = i - 1;

        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
}

void *sortThread(void *arg) {
    SortData *data = (SortData *)arg;
    clock_t start, end;
    double time_taken;

    start = clock();

    if (strcmp(data->name, "Bubble") == 0)
        bubbleSort(data->arr, data->n);
    else if (strcmp(data->name, "Selection") == 0)
        selectionSort(data->arr, data->n);
    else if (strcmp(data->name, "Insertion") == 0)
        insertionSort(data->arr, data->n);

    end = clock();
    time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;

    printf("%s Sort Result: ", data->name);
    printArray(data->arr, data->n);
    printf("%s Sort Time Taken: %f seconds\n", data->name, time_taken);

    pthread_exit(NULL);
}

int main(int argc, char *argv[]) {
    if (argc < 3) {
        printf("Usage: %s <n> <elements>\n", argv[0]);
        return 1;
    }

    int n = atoi(argv[1]);

    if (argc != n + 2) {
        printf("Please enter exactly %d elements.\n", n);
        return 1;
    }

    int *original = (int *)malloc(n * sizeof(int));
    int *arr1 = (int *)malloc(n * sizeof(int));
    int *arr2 = (int *)malloc(n * sizeof(int));
    int *arr3 = (int *)malloc(n * sizeof(int));

    for (int i = 0; i < n; i++) {
        original[i] = atoi(argv[i + 2]);
        arr1[i] = original[i];
        arr2[i] = original[i];
        arr3[i] = original[i];
    }

    pthread_t t1, t2, t3;

    SortData d1 = {arr1, n, "Bubble"};
    SortData d2 = {arr2, n, "Selection"};
    SortData d3 = {arr3, n, "Insertion"};

    pthread_create(&t1, NULL, sortThread, &d1);
    pthread_create(&t2, NULL, sortThread, &d2);
    pthread_create(&t3, NULL, sortThread, &d3);

    pthread_join(t1, NULL);
    pthread_join(t2, NULL);
    pthread_join(t3, NULL);

    printf("Final Sorted Array (using Bubble result): ");
    printArray(arr1, n);

    free(original);
    free(arr1);
    free(arr2);
    free(arr3);

    return 0;
}

5. Write multithreaded program that generates the Fibonacci series. The program should work as follows: The user will enter on the command line the number of Fibonacci numbers that the program is to generate. The program then will create a separate thread that will generate the Fibonacci numbers, placing the sequence in data that is shared by the threads. When the thread finishes execution the parent will output the sequence generated by the child thread. Because the parent thread cannot begin outputting the Fibonacci sequence until the child thread finishes, this will require having the parent thread wait for the child thread to finish.



#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>

int *fib;
int n;

void *generateFib(void *arg) {
    if (n >= 1)
        fib[0] = 0;
    if (n >= 2)
        fib[1] = 1;

    for (int i = 2; i < n; i++) {
        fib[i] = fib[i - 1] + fib[i - 2];
    }

    pthread_exit(NULL);
}

int main(int argc, char *argv[]) {
    if (argc != 2) {
        printf("Usage: %s <number_of_terms>\n", argv[0]);
        return 1;
    }

    n = atoi(argv[1]);

    if (n <= 0) {
        printf("Enter a positive number\n");
        return 1;
    }

    fib = (int *)malloc(n * sizeof(int));
    pthread_t tid;

    pthread_create(&tid, NULL, generateFib, NULL);
    pthread_join(tid, NULL);

    printf("Fibonacci Series: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", fib[i]);
    }
    printf("\n");

    free(fib);
    return 0;
}

2. SRTF


#include <stdio.h>
#include <limits.h>
#include <string.h>

struct Gantt {
    char pid[3];
    int start;
    int end;
};

int main() {
    int n = 4;
    char p[4][3] = {"P1", "P2", "P3", "P4"};
    int at[4] = {0, 3, 4, 9};
    int bt[4] = {60, 30, 40, 10};
    int rt[4], ct[4] = {0}, tat[4], wt[4];

    struct Gantt g[100];
    int gcount = 0;

    for (int i = 0; i < n; i++) rt[i] = bt[i];

    int completed = 0, time = 0, prev = -1;
    float avg_tat = 0, avg_wt = 0;

    while (completed < n) {
        int idx = -1;
        int min_rt = INT_MAX;

        for (int i = 0; i < n; i++) {
            if (at[i] <= time && rt[i] > 0) {
                if (rt[i] < min_rt) {
                    min_rt = rt[i];
                    idx = i;
                }
                else if (rt[i] == min_rt) {
                    if (at[i] < at[idx]) idx = i;
                }
            }
        }

        if (idx == -1) {
            time++;
            continue;
        }

        if (prev != idx) {
            strcpy(g[gcount].pid, p[idx]);
            g[gcount].start = time;
            if (gcount > 0) g[gcount - 1].end = time;
            gcount++;
        }

        rt[idx]--;
        time++;
        prev = idx;

        if (rt[idx] == 0) {
            ct[idx] = time;
            completed++;
        }
    }

    if (gcount > 0) g[gcount - 1].end = time;

    for (int i = 0; i < n; i++) {
        tat[i] = ct[i] - at[i];
        wt[i] = tat[i] - bt[i];
        avg_tat += tat[i];
        avg_wt += wt[i];
    }

    printf("\nSRTF Scheduling\n");

    printf("\nGantt Chart:\n");
    for (int i = 0; i < gcount; i++) printf("--------");
    printf("-\n|");
    for (int i = 0; i < gcount; i++) printf("  %s  |", g[i].pid);
    printf("\n");
    for (int i = 0; i < gcount; i++) printf("--------");
    printf("-\n");

    printf("%d", g[0].start);
    for (int i = 0; i < gcount; i++) printf("      %d", g[i].end);
    printf("\n");

    printf("\n%-10s %-10s %-10s %-15s %-15s %-12s\n",
           "Process", "AT", "BT", "CT", "TAT", "WT");

    for (int i = 0; i < n; i++) {
        printf("%-10s %-10d %-10d %-15d %-15d %-12d\n",
               p[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time = %.2f\n", avg_tat / n);
    printf("Average Waiting Time    = %.2f\n", avg_wt / n);

    return 0;
}

3. Round Robin

#include <stdio.h>
#include <string.h>

struct Gantt {
    char pid[3];
    int start;
    int end;
};

int main() {
    int n = 4, tq = 10;
    char p[4][3] = {"P1", "P2", "P3", "P4"};
    int at[4] = {0, 3, 4, 9};
    int bt[4] = {60, 30, 40, 10};
    int rt[4], ct[4] = {0}, tat[4], wt[4];
    int visited[4] = {0};

    int q[100], front = 0, rear = 0;
    struct Gantt g[100];
    int gcount = 0;

    for (int i = 0; i < n; i++) rt[i] = bt[i];

    int time = 0, completed = 0;
    float avg_tat = 0, avg_wt = 0;

    while (completed < n) {
        for (int i = 0; i < n; i++) {
            if (!visited[i] && at[i] <= time) {
                q[rear++] = i;
                visited[i] = 1;
            }
        }

        if (front == rear) {
            time++;
            continue;
        }

        int idx = q[front++];

        strcpy(g[gcount].pid, p[idx]);
        g[gcount].start = time;

        if (rt[idx] > tq) {
            time += tq;
            rt[idx] -= tq;
        } else {
            time += rt[idx];
            rt[idx] = 0;
            ct[idx] = time;
            completed++;
        }

        g[gcount].end = time;
        gcount++;

        for (int i = 0; i < n; i++) {
            if (!visited[i] && at[i] <= time) {
                q[rear++] = i;
                visited[i] = 1;
            }
        }

        if (rt[idx] > 0) {
            q[rear++] = idx;
        }
    }

    for (int i = 0; i < n; i++) {
        tat[i] = ct[i] - at[i];
        wt[i] = tat[i] - bt[i];
        avg_tat += tat[i];
        avg_wt += wt[i];
    }

    printf("\nRound Robin Scheduling (Time Quantum = 10)\n");

    printf("\nGantt Chart:\n");
    for (int i = 0; i < gcount; i++) printf("--------");
    printf("-\n|");
    for (int i = 0; i < gcount; i++) printf("  %s  |", g[i].pid);
    printf("\n");
    for (int i = 0; i < gcount; i++) printf("--------");
    printf("-\n");

    printf("%d", g[0].start);
    for (int i = 0; i < gcount; i++) printf("      %d", g[i].end);
    printf("\n");

    printf("\n%-10s %-10s %-10s %-15s %-15s %-12s\n",
           "Process", "AT", "BT", "CT", "TAT", "WT");

    for (int i = 0; i < n; i++) {
        printf("%-10s %-10d %-10d %-15d %-15d %-12d\n",
               p[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time = %.2f\n", avg_tat / n);
    printf("Average Waiting Time    = %.2f\n", avg_wt / n);

    return 0;
}

4. Non-preemptive priority


#include <stdio.h>
#include <string.h>

int main() {
    int n = 4;
    char p[4][3] = {"P1", "P2", "P3", "P4"};
    int at[4] = {0, 3, 4, 9};
    int bt[4] = {60, 30, 40, 10};
    int pr[4] = {3, 2, 1, 4};
    int ct[4] = {0}, tat[4], wt[4], done[4] = {0};

    char gantt_pid[10][3];
    int start[10], end[10], gcount = 0;

    int completed = 0, time = 0;
    float avg_tat = 0, avg_wt = 0;

    while (completed < n) {
        int idx = -1;
        int highest = -1;

        for (int i = 0; i < n; i++) {
            if (!done[i] && at[i] <= time) {
                if (pr[i] > highest) {
                    highest = pr[i];
                    idx = i;
                } else if (pr[i] == highest) {
                    if (at[i] < at[idx]) idx = i;
                }
            }
        }

        if (idx == -1) {
            time++;
            continue;
        }

        strcpy(gantt_pid[gcount], p[idx]);
        start[gcount] = time;
        time += bt[idx];
        end[gcount] = time;
        gcount++;

        ct[idx] = time;
        done[idx] = 1;
        completed++;
    }

    for (int i = 0; i < n; i++) {
        tat[i] = ct[i] - at[i];
        wt[i] = tat[i] - bt[i];
        avg_tat += tat[i];
        avg_wt += wt[i];
    }

    printf("\nNon-Preemptive Priority Scheduling\n");
    printf("(Higher number = Higher priority)\n");

    printf("\nGantt Chart:\n");
    for (int i = 0; i < gcount; i++) printf("--------");
    printf("-\n|");
    for (int i = 0; i < gcount; i++) printf("  %s  |", gantt_pid[i]);
    printf("\n");
    for (int i = 0; i < gcount; i++) printf("--------");
    printf("-\n");

    printf("%d", start[0]);
    for (int i = 0; i < gcount; i++) printf("      %d", end[i]);
    printf("\n");

    printf("\n%-10s %-10s %-10s %-15s %-15s %-12s\n",
           "Process", "AT", "BT", "CT", "TAT", "WT");

    for (int i = 0; i < n; i++) {
        printf("%-10s %-10d %-10d %-15d %-15d %-12d\n",
               p[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time = %.2f\n", avg_tat / n);
    printf("Average Waiting Time    = %.2f\n", avg_wt / n);

    return 0;
}
